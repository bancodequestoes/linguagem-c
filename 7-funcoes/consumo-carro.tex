\question[10]

\begin{table}[h!]
	\centering
	\begin{tabular}{|c|c|c|}
		\hline 
		\textbf{Consumo} & \textbf{Km/l} & \textbf{Mensagem} \\ 
		\hline 
		Menor que & 8 & Venda o Carro! \\ 
		\hline 
		Entre & 8 e 12 & Econômico \\ 
		\hline 
		Maior que & 12 & Super Econômico! \\ 
		\hline 
	\end{tabular} 
\end{table}

\question[10]

Qual valor a variável \textbf{x} tem no final desse trecho de código? Marque a opção correta.

\begin{lstlisting}
#include<stdio.h>

	int main(){
	
	int foo = 12;
	int bar = 0;
	int i,x = 0;
	
	if(foo % 4 < 10){
		bar = 2;
	}else{
		bar = 1;
	}
	
	if(foo % 3 < 10){
		bar = 3;
	}
	
	for(i=0; i<bar; i++){
		x = x + 2;
	}
	
	printf("%d", x);
}

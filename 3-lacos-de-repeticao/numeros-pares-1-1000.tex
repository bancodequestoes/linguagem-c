\question[10]

Escreva um programa em C que imprime na tela todos os números pares de 1 até 1000 (inclusive).

\begin{solution}
\begin{lstlisting}
#include <stdio.h>

int main() {

	int i;
	
	for(i=1; i<=1000; i++){
		if(i % 2 == 0){
			printf("%d \n", i);
		}
	}
}
\end{lstlisting}
\end{solution}
